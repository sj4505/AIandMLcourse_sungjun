```python
"""
Week 4: Neural Network Labs — PySide6 GUI
탭 기반 통합 앱. 각 Lab의 학습을 QThread에서 실행하고
FigureCanvas로 결과를 앱 안에 표시합니다.

import sys
from pathlib import Path

os.environ['TF_CPP_MIN_LOG_LEVEL'] = '2'
import matplotlib
matplotlib.use('QtAgg')

import numpy as np
import matplotlib.pyplot as plt
from matplotlib.figure import Figure
from matplotlib.backends.backend_qtagg import FigureCanvasQTAgg as FigureCanvas

from PySide6.QtWidgets import (
    QApplication, QMainWindow, QWidget, QTabWidget,
    QSplitter, QVBoxLayout, QHBoxLayout, QFormLayout,
    QLabel, QPushButton, QProgressBar, QTextEdit,
    QSpinBox, QDoubleSpinBox, QLineEdit, QComboBox,
    QMessageBox, QScrollArea, QSizePolicy,
)
from PySide6.QtCore import Qt, QThread, Signal
from PySide6.QtGui import QFont

# 출력 디렉토리: week4/outputs/
OUTPUT_DIR = Path(__file__).parent / 'outputs'
OUTPUT_DIR.mkdir(exist_ok=True)

try:
    import tensorflow as tf
    from tensorflow import keras
    TF_AVAILABLE = True
except ImportError:
    TF_AVAILABLE = False


# ---------------------------------------------------------------------------
# Keras callback: epoch마다 signal emit
# ---------------------------------------------------------------------------

class SignalCallback(keras.callbacks.Callback):
    """학습 진행 상황을 QThread signal로 전달하는 Keras Callback."""

    def __init__(self, total_epochs, progress_signal, log_signal):
        super().__init__()
        self.total_epochs = total_epochs
        self.progress_signal = progress_signal
        self.log_signal = log_signal

    def on_epoch_end(self, epoch, logs=None):
        pct = int((epoch + 1) / self.total_epochs * 100)
        self.progress_signal.emit(pct)
        loss = logs.get('loss', 0.0)
        val_loss = logs.get('val_loss', None)
        if val_loss is not None:
            msg = f"Epoch {epoch+1}/{self.total_epochs} — loss: {loss:.6f}  val_loss: {val_loss:.6f}"
        else:
            msg = f"Epoch {epoch+1}/{self.total_epochs} — loss: {loss:.6f}"
        self.log_signal.emit(msg)


# ---------------------------------------------------------------------------
# TrainingWorker: QThread로 학습 실행
# ---------------------------------------------------------------------------

class TrainingWorker(QThread):
    """백그라운드에서 학습 함수를 실행하는 QThread."""

    progress = Signal(int)          # 0~100
    log_message = Signal(str)       # 에포크 로그
    finished = Signal(list)         # list[Figure]
    error = Signal(str)             # 에러 메시지

    def __init__(self, train_func, params):
        super().__init__()
        self.train_func = train_func
        self.params = params

    def run(self):
        try:
            figures = self.train_func(
                self.params,
                self.progress,
                self.log_message,
            )
            self.finished.emit(figures)
        except Exception as exc:
            self.error.emit(str(exc))


# ---------------------------------------------------------------------------
# 파라미터 파싱 유틸
# ---------------------------------------------------------------------------

def parse_int_list(text):
    """'128, 128, 64' → [128, 128, 64]. 실패시 None 반환."""
    try:
        return [int(x.strip()) for x in text.split(',') if x.strip()]
    except ValueError:
        return None


def parse_float_list(text):
    """'0.5, 1.0, 2.0' → [0.5, 1.0, 2.0]. 실패시 None 반환."""
    try:
        return [float(x.strip()) for x in text.split(',') if x.strip()]
    except ValueError:
        return None


if __name__ == '__main__':
    app = QApplication(sys.argv)
    # MainWindow will be added in Task 8
    sys.exit(app.exec())
```

```python
# ---------------------------------------------------------------------------
# BaseLabTab: 모든 Lab 탭의 공통 레이아웃

class BaseLabTab(QWidget):
    """
    왼쪽: 파라미터 패널 (QFormLayout + 실행 버튼)
    오른쪽: 진행률 바 + 로그 + FigureCanvas 스크롤 영역
    """

    # 전역 학습 중 여부 — 모든 탭의 실행 버튼 비활성화용
    _training_active = False

    def __init__(self, title, parent=None):
        super().__init__(parent)
        self.title = title
        self.worker = None
        self._setup_ui()

    def _setup_ui(self):
        main_layout = QHBoxLayout(self)
        splitter = QSplitter(Qt.Horizontal)
        main_layout.addWidget(splitter)

        # ── 왼쪽: 파라미터 패널 ──────────────────────────────────────
        left_widget = QWidget()
        left_widget.setMaximumWidth(220)
        left_layout = QVBoxLayout(left_widget)

        param_label = QLabel("파라미터")
        param_label.setFont(QFont('Arial', 9))
        param_label.setStyleSheet("color: gray; text-transform: uppercase;")
        left_layout.addWidget(param_label)

        self.form = QFormLayout()
        self.form.setSpacing(8)
        left_layout.addLayout(self.form)

        # 서브클래스에서 폼 위젯 채움
        self._add_params(self.form)

        left_layout.addStretch()

        self.run_btn = QPushButton("▶ 실행")
        self.run_btn.setMinimumHeight(36)
        self.run_btn.clicked.connect(self._on_run_clicked)
        left_layout.addWidget(self.run_btn)

        splitter.addWidget(left_widget)

        # ── 오른쪽: 결과 패널 ────────────────────────────────────────
        right_widget = QWidget()
        right_layout = QVBoxLayout(right_widget)

        # 진행률
        progress_row = QHBoxLayout()
        self.progress_bar = QProgressBar()
        self.progress_bar.setRange(0, 100)
        self.progress_bar.setValue(0)
        self.progress_label = QLabel("대기 중")
        self.progress_label.setFixedWidth(60)
        progress_row.addWidget(self.progress_bar)
        progress_row.addWidget(self.progress_label)
        right_layout.addLayout(progress_row)

        # 로그
        self.log_edit = QTextEdit()
        self.log_edit.setReadOnly(True)
        self.log_edit.setMaximumHeight(100)
        self.log_edit.setFont(QFont('Courier New', 9))
        self.log_edit.setStyleSheet("background: #1e1e1e; color: #d4d4d4;")
        right_layout.addWidget(self.log_edit)

        # FigureCanvas 스크롤 영역
        self.scroll = QScrollArea()
        self.scroll.setWidgetResizable(True)
        self.canvas_container = QWidget()
        self.canvas_layout = QVBoxLayout(self.canvas_container)
        self.scroll.setWidget(self.canvas_container)
        right_layout.addWidget(self.scroll)

        splitter.addWidget(right_widget)
        splitter.setSizes([220, 700])

    def _add_params(self, form: QFormLayout):
        """서브클래스에서 override: form에 파라미터 위젯 추가."""
        raise NotImplementedError

    def _collect_params(self) -> dict | None:
        """서브클래스에서 override: 파라미터 dict 반환. 검증 실패 시 None."""
        raise NotImplementedError

    def _train_func(self, params, progress_signal, log_signal) -> list:
        """서브클래스에서 override: 학습 실행 후 list[Figure] 반환."""
        raise NotImplementedError

    def _on_run_clicked(self):
        if not TF_AVAILABLE:
            QMessageBox.critical(self, "에러", "TensorFlow를 찾을 수 없습니다.\n`uv sync` 또는 `pip install tensorflow`를 실행해주세요.")
            return

        if BaseLabTab._training_active:
            QMessageBox.information(self, "학습 중", "다른 Lab이 학습 중입니다. 완료 후 실행해주세요.")
            return

        params = self._collect_params()
        if params is None:
            return  # 검증 실패 — _collect_params 안에서 QMessageBox 표시

        # UI 초기화
        self.log_edit.clear()
        self.progress_bar.setValue(0)
        self.progress_label.setText("0%")
        # 기존 캔버스 제거
        for i in reversed(range(self.canvas_layout.count())):
            self.canvas_layout.itemAt(i).widget().deleteLater()

        # 전역 실행 중 플래그
        BaseLabTab._training_active = True
        self._set_all_run_buttons(False)

        self.worker = TrainingWorker(self._train_func, params)
        self.worker.progress.connect(self._on_progress)
        self.worker.log_message.connect(self._on_log)
        self.worker.finished.connect(self._on_finished)
        self.worker.error.connect(self._on_error)
        self.worker.start()

    def _on_progress(self, pct):
        self.progress_bar.setValue(pct)
        self.progress_label.setText(f"{pct}%")

    def _on_log(self, msg):
        self.log_edit.append(msg)
        self.log_edit.verticalScrollBar().setValue(
            self.log_edit.verticalScrollBar().maximum()
        )

    def _on_finished(self, figures):
        BaseLabTab._training_active = False
        self._set_all_run_buttons(True)
        self.progress_bar.setValue(100)
        self.progress_label.setText("완료")
        for fig in figures:
            canvas = FigureCanvas(fig)
            canvas.setSizePolicy(QSizePolicy.Expanding, QSizePolicy.Fixed)
            canvas.setMinimumHeight(400)
            self.canvas_layout.addWidget(canvas)

    def _on_error(self, msg):
        BaseLabTab._training_active = False
        self._set_all_run_buttons(True)
        self.progress_label.setText("오류")
        QMessageBox.critical(self, "학습 오류", f"학습 중 오류 발생:\n{msg}")

    def _set_all_run_buttons(self, enabled: bool):
        """최상위 창에서 모든 탭의 실행 버튼 활성화/비활성화."""
        main_win = self.window()
        if not isinstance(main_win, MainWindow):
            return
        tab_widget = main_win.tab_widget
        for i in range(tab_widget.count()):
            tab = tab_widget.widget(i)
            if isinstance(tab, BaseLabTab):
                tab.run_btn.setEnabled(enabled)
```

```python
# ---------------------------------------------------------------------------
# Lab 1: 1D 함수 근사
# ---------------------------------------------------------------------------

def train_lab1(params, progress_signal, log_signal):
    """
    1D 함수 근사 학습.
    params: {epochs, hidden_layers, activation, lr}
    반환: list[Figure] — [approximation, network_comparison, extreme_test]
    """
    epochs = params['epochs']
    hidden_layers = params['hidden_layers']
    activation = params['activation']
    lr = params['lr']

    figures = []

    def make_model(layers, act=activation, learning_rate=lr):
        model = keras.Sequential()
        model.add(keras.layers.Input(shape=(1,)))
        for units in layers:
            model.add(keras.layers.Dense(units, activation=act))
        model.add(keras.layers.Dense(1, activation='linear'))
        model.compile(optimizer=keras.optimizers.Adam(learning_rate=learning_rate), loss='mse')
        return model

    cb = SignalCallback(epochs, progress_signal, log_signal)

    # ── 1. 3개 함수 근사 ─────────────────────────────────────────────
    x_train = np.linspace(-2*np.pi, 2*np.pi, 200).reshape(-1, 1)
    x_test  = np.linspace(-2*np.pi, 2*np.pi, 400).reshape(-1, 1)

    funcs = {
        'sin(x)': (np.sin(x_train), np.sin(x_test)),
        'cos(x)+0.5sin(2x)': (np.cos(x_train)+0.5*np.sin(2*x_train),
                               np.cos(x_test)+0.5*np.sin(2*x_test)),
        'x·sin(x)': (x_train*np.sin(x_train), x_test*np.sin(x_test)),
    }

    fig1 = Figure(figsize=(18, 12))
    axes = fig1.subplots(3, 3)

    for idx, (name, (y_tr, y_te)) in enumerate(funcs.items()):
        log_signal.emit(f"[Lab1] 학습 중: {name}")
        # 각 함수마다 별도 epoch 비율로 progress 분배
        class PartialCallback(keras.callbacks.Callback):
            def __init__(self, offset, total_funcs):
                super().__init__()
                self.offset = offset
                self.total_funcs = total_funcs
            def on_epoch_end(self, epoch, logs=None):
                pct = int((self.offset + (epoch+1)/epochs) / self.total_funcs * 100)
                progress_signal.emit(pct)
                loss = logs.get('loss', 0)
                log_signal.emit(f"Epoch {epoch+1}/{epochs} loss:{loss:.6f}")

        pcb = PartialCallback(idx, 3)
        m = make_model(hidden_layers)
        m.fit(x_train, y_tr, epochs=epochs, batch_size=32, verbose=0, callbacks=[pcb])
        y_pred = m.predict(x_test, verbose=0)
        mse = float(np.mean((y_pred - y_te)**2))

        ax0 = axes[idx, 0]
        ax0.plot(x_test, y_te, 'b-', lw=2, label='True', alpha=0.7)
        ax0.plot(x_test, y_pred, 'r--', lw=2, label='Predicted')
        ax0.set_title(f'{name}\nMSE:{mse:.6f}', fontsize=11, fontweight='bold')
        ax0.legend(fontsize=8); ax0.grid(True, alpha=0.3)

        ax1 = axes[idx, 1]
        hist = m.history.history
        ax1.plot(hist['loss'], 'g-', lw=1.5)
        ax1.set_title('Training Loss', fontsize=11); ax1.set_yscale('log')
        ax1.set_xlabel('Epoch'); ax1.grid(True, alpha=0.3)

        ax2 = axes[idx, 2]
        err = np.abs(y_pred - y_te)
        ax2.plot(x_test, err, 'r-', lw=1.5)
        ax2.fill_between(x_test.flatten(), 0, err.flatten(), color='r', alpha=0.3)
        ax2.set_title(f'Error (max:{float(err.max()):.4f})', fontsize=11)
        ax2.grid(True, alpha=0.3)

    fig1.suptitle('1D 함수 근사 (TensorFlow/Keras)', fontsize=14, fontweight='bold')
    fig1.tight_layout()
    fig1.savefig(OUTPUT_DIR / 'perfect_1d_approximation.png', dpi=150, bbox_inches='tight')
    figures.append(fig1)

    # ── 2. 네트워크 크기 비교 ─────────────────────────────────────────
    log_signal.emit("[Lab1] 네트워크 크기 비교 중...")
    x = np.linspace(-2*np.pi, 2*np.pi, 200).reshape(-1, 1)
    y = x * np.sin(x) + 0.3 * np.cos(3*x)

    arch_map = {
        'Small [32]': [32],
        'Medium [64,64]': [64, 64],
        'Large [128,128]': [128, 128],
        'VeryLarge [128,128,64]': [128, 128, 64],
    }

    fig2 = Figure(figsize=(14, 10))
    axs = fig2.subplots(2, 2)
    axs = axs.flatten()

    for i, (name, layers) in enumerate(arch_map.items()):
        m = make_model(layers)
        m.fit(x, y, epochs=min(epochs, 2000), batch_size=32, verbose=0)
        y_pred = m.predict(x, verbose=0)
        mse = float(np.mean((y_pred - y)**2))
        axs[i].plot(x, y, 'b-', lw=2, label='True', alpha=0.7)
        axs[i].plot(x, y_pred, 'r--', lw=2, label='Predicted')
        axs[i].set_title(f'{name}\nMSE:{mse:.6f}', fontsize=11, fontweight='bold')
        axs[i].legend(fontsize=8); axs[i].grid(True, alpha=0.3)

    fig2.suptitle('네트워크 크기 비교', fontsize=14, fontweight='bold')
    fig2.tight_layout()
    fig2.savefig(OUTPUT_DIR / 'network_size_comparison.png', dpi=150, bbox_inches='tight')
    figures.append(fig2)

    # ── 3. 극한 복잡도 테스트 ─────────────────────────────────────────
    log_signal.emit("[Lab1] 극한 복잡도 테스트 중...")
    x2 = np.linspace(-3*np.pi, 3*np.pi, 500).reshape(-1, 1)
    y2 = (np.sin(x2) + 0.5*np.sin(2*x2) + 0.3*np.cos(3*x2)
          + 0.2*np.sin(5*x2) + 0.1*x2*np.cos(x2))

    m3 = make_model([256, 256, 128, 64])
    lr_cb = keras.callbacks.ReduceLROnPlateau(monitor='loss', factor=0.8, patience=100, min_lr=1e-6, verbose=0)
    es_cb = keras.callbacks.EarlyStopping(monitor='loss', patience=500, restore_best_weights=True, verbose=0)
    m3.fit(x2, y2, epochs=min(epochs*2, 8000), batch_size=32, verbose=0, callbacks=[lr_cb, es_cb])
    y3_pred = m3.predict(x2, verbose=0)
    mse3 = float(np.mean((y3_pred - y2)**2))
    err3 = np.abs(y3_pred - y2)

    fig3 = Figure(figsize=(14, 10))
    ax30, ax31, ax32, ax33 = fig3.subplots(2, 2).flatten()
    ax30.plot(x2, y2, 'b-', lw=2, label='True', alpha=0.7)
    ax30.plot(x2, y3_pred, 'r--', lw=2, label='NN')
    ax30.set_title(f'Extreme Function\nMSE:{mse3:.8f}', fontsize=12); ax30.legend(); ax30.grid(True, alpha=0.3)
    ax31.plot(m3.history.history['loss'], 'g-', lw=1.5)
    ax31.set_title('Training Loss'); ax31.set_yscale('log'); ax31.grid(True, alpha=0.3)
    ax32.plot(x2, err3, 'r-', lw=1.5)
    ax32.fill_between(x2.flatten(), 0, err3.flatten(), color='r', alpha=0.3)
    ax32.set_title(f'Error (max:{float(err3.max()):.4f})'); ax32.grid(True, alpha=0.3)
    ax33.hist(err3.flatten(), bins=40, color='r', alpha=0.7, edgecolor='black')
    ax33.set_title('Error Distribution'); ax33.grid(True, alpha=0.3, axis='y')

    fig3.suptitle('극한 복잡도 테스트', fontsize=14, fontweight='bold')
    fig3.tight_layout()
    fig3.savefig(OUTPUT_DIR / 'extreme_function_test.png', dpi=150, bbox_inches='tight')
    figures.append(fig3)

    progress_signal.emit(100)
    return figures


class Lab1Tab(BaseLabTab):
    def __init__(self, parent=None):
        super().__init__('Lab 1: 1D 함수 근사', parent)

    def _add_params(self, form):
        self.epochs_spin = QSpinBox()

        self.epochs_spin.setValue(3000)
        form.addRow('Epochs', self.epochs_spin)

        self.layers_edit = QLineEdit('128, 128, 64')
        form.addRow('Hidden Layers', self.layers_edit)

        self.act_combo = QComboBox()

    def _collect_params(self):
```

```python
def train_lab4(params, progress_signal, log_signal):

        t0 = np.deg2rad(theta0_deg)
        T_small = 2 * np.pi * np.sqrt(L / g)
        correction = 1 + (1/16)*t0**2 + (11/3072)*t0**4
        return T_small * correction

    # 데이터 생성
    log_signal.emit("[Lab4] 데이터 생성 중...")
    n = 2000
    L_data = np.random.uniform(0.5, 3.0, n)
    th_data = np.random.uniform(5, 80, n)
    T_data = true_period(L_data, th_data) * (1 + np.random.normal(0, 0.01, n))

    Y = T_data.reshape(-1, 1)

    model = keras.Sequential([
        keras.layers.Input(shape=(2,)),
        keras.layers.Dense(64, activation='relu'),
        keras.layers.Dropout(0.1),
        keras.layers.Dense(32, activation='relu'),
        keras.layers.Dropout(0.1),
        keras.layers.Dense(16, activation='relu'),
        keras.layers.Dropout(0.1),
        keras.layers.Dense(1, activation='linear'),
    ])
    model.compile(optimizer=keras.optimizers.Adam(0.001), loss='mse',
                  metrics=['mae'])

    cb = SignalCallback(epochs, progress_signal, log_signal)
    model.fit(X, Y, validation_split=0.2, epochs=epochs,
              batch_size=32, verbose=0, callbacks=[cb])

    # ── 그래프 1: 길이별 주기 예측 ────────────────────────────────────

    fig1 = Figure(figsize=(14, 5))

    for i, L in enumerate(lengths):
        T_true = true_period(L, th_range)
        Xi = np.column_stack([np.full(100, L), th_range])
        T_pred = model.predict(Xi, verbose=0).flatten()
        mape = float(np.mean(np.abs((T_pred - T_true) / T_true)) * 100)

        axes1[i].plot(th_range, T_true, 'b-', lw=2, label='True', alpha=0.7)

    # ── 그래프 2: RK4 시뮬레이션 ──────────────────────────────────────
    def rk4_pendulum(L, theta0_deg, t_end=10.0, dt=0.01):
        theta0 = np.deg2rad(theta0_deg)
        t = np.arange(0, t_end, dt)
        theta = np.zeros(len(t))
        omega = np.zeros(len(t))
        theta[0] = theta0
        for k in range(len(t) - 1):
            def f(th, om):
                return om, -(g/L)*np.sin(th)
            k1t, k1o = f(theta[k], omega[k])
            k2t, k2o = f(theta[k]+dt/2*k1t, omega[k]+dt/2*k1o)
            k3t, k3o = f(theta[k]+dt/2*k2t, omega[k]+dt/2*k2o)
            k4t, k4o = f(theta[k]+dt*k3t, omega[k]+dt*k3o)
            theta[k+1] = theta[k] + dt*(k1t+2*k2t+2*k3t+k4t)/6
            omega[k+1] = omega[k] + dt*(k1o+2*k2o+2*k3o+k4o)/6
        return t, theta, omega

    fig2 = Figure(figsize=(14, 4*len(angles)))
    axes2 = fig2.subplots(len(angles), 2)

        axes2[i][0].set_xlabel('t (s)'); axes2[i][0].set_ylabel('θ (°)')
        axes2[i][0].grid(True, alpha=0.3)

        axes2[i][1].plot(np.rad2deg(th_), np.rad2deg(om_), 'r-', lw=0.8, alpha=0.7)
        axes2[i][1].set_title('위상 공간 (θ vs ω)', fontsize=11)
        axes2[i][1].set_xlabel('θ (°)'); axes2[i][1].set_ylabel('ω (°/s)')
        axes2[i][1].grid(True, alpha=0.3)

    fig2.tight_layout()

    figures.append(fig2)

    # ── 그래프 3: 학습 곡선 + 오차 분석 ──────────────────────────────
    hist = model.history.history
    fig3 = Figure(figsize=(14, 10))
    ax1, ax2, ax3, ax4 = fig3.subplots(2, 2).flatten()

    ax1.plot(hist['loss'], 'b-', lw=2, label='Train')
    ax1.plot(hist['val_loss'], 'r--', lw=2, label='Val')
    ax1.set_title('Training History'); ax1.legend(); ax1.set_yscale('log'); ax1.grid(True, alpha=0.3)

    ax2.plot(hist['mae'], 'b-', lw=2, label='Train MAE')
    ax2.plot(hist['val_mae'], 'r--', lw=2, label='Val MAE')
    ax2.set_title('MAE'); ax2.legend(); ax2.grid(True, alpha=0.3)

    L_range = np.linspace(0.5, 3.0, 50)
    err_L = []
    for Li in L_range:
        Xi = np.column_stack([np.full(50, Li), np.linspace(5, 80, 50)])
        T_p = model.predict(Xi, verbose=0).flatten()
        T_t = true_period(Li, np.linspace(5, 80, 50))
        err_L.append(float(np.mean(np.abs((T_p-T_t)/T_t))*100))
    ax3.plot(L_range, err_L, 'go-', lw=2, markersize=6)
    ax3.set_title('MAPE vs Length'); ax3.set_xlabel('L (m)'); ax3.set_ylabel('MAPE (%)'); ax3.grid(True, alpha=0.3)

    th_range2 = np.linspace(5, 80, 50)

        T_t = true_period(1.0, th_i)
        err_th.append(float(abs((T_p[0]-T_t)/T_t)*100))
    ax4.plot(th_range2, err_th, 'mo-', lw=2, markersize=6)
```
